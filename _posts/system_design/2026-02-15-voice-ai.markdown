## Voice AI Interactive Demo

<div style="margin: 40px 0; text-align: center;">
    <button id="test-voice-btn" style="
        background-color: #333; 
        color: #ffffff; 
        border: none; 
        padding: 20px 40px; 
        font-size: 1.2rem; 
        font-weight: 600; 
        border-radius: 8px; 
        cursor: wait; 
        transition: all 0.3s ease;
    ">
        Initializing Engine...
    </button>
</div>

<script src="https://unpkg.com/@elevenlabs/convai-sdk@latest/dist/index.bundle.js"></script>

<script>
  (function() {
    const btn = document.getElementById('test-voice-btn');
    let conversation;

    // 2. This checks for the SDK every 100ms
    const timer = setInterval(() => {
      if (window.ElevenLabsConvAI) {
        clearInterval(timer);
        btn.textContent = "Click to Speak with AI";
        btn.style.backgroundColor = "#00D1FF";
        btn.style.cursor = "pointer";
        console.log("SDK Ready");
      }
    }, 100);

    btn.onclick = async () => {
      if (!window.ElevenLabsConvAI) return;

      if (!conversation) {
        try {
          btn.textContent = "Connecting...";
          
          // 3. Start the actual session
          conversation = await window.ElevenLabsConvAI.Conversation.startSession({
            agentId: "agent_4601khceyk26ebb9rbdrx4kvbn4b",
            onConnect: () => {
              btn.textContent = "Stop AI";
              btn.style.backgroundColor = "#FF3B30";
            },
            onDisconnect: () => {
              conversation = null;
              btn.textContent = "Click to Speak with AI";
              btn.style.backgroundColor = "#00D1FF";
            }
          });
        } catch (err) {
          console.error(err);
          btn.textContent = "Mic Error - Check Settings";
        }
      } else {
        await conversation.endSession();
      }
    };
  })();
</script>

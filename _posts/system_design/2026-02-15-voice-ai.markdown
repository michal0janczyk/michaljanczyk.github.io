btn.onclick = async () => {
  console.log("Button clicked..."); 
  try {
    if (!window.ElevenLabsConvAI) {
      alert("SDK not loaded yet!");
      return;
    }
    
    btn.textContent = "Connecting...";
    
    conversation = await window.ElevenLabsConvAI.Conversation.startSession({
      agentId: "agent_4601khceyk26ebb9rbdrx4kvbn4b",
      onConnect: () => {
        alert("Connected!"); // If this pops up, it works!
        btn.textContent = "Stop AI";
      },
      onDisconnect: () => {
        btn.textContent = "Click to Speak";
      },
      onError: (err) => {
        alert("Error: " + JSON.stringify(err));
      }
    });
  } catch (err) {
    alert("Fatal Error: " + err.message);
  }
};
